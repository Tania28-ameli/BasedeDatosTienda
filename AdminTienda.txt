namespace PresentacionTienda
{
    public partial class FrmMostrarProductos : Form
    {
        ManejadorProductos mp;
        public static int fila = 0, columna = 0;
        public static string nombre = "", descripcion="", precio="";

        private void textBox1_TextChanged(object sender, EventArgs e)
        {
           dataGridView1.Visible = true;
            mp.Mostrar(dataGridView1, txtBuscar.Text);

        }
        void limpiar()
        {
            dataGridView1.Visible = false;
        }

        public FrmMostrarProductos()
        {
            InitializeComponent();
            mp= new ManejadorProductos();
             
    }

        private void dataGridView1_CellEnter(object sender, DataGridViewCellEventArgs e)
        {
            fila = e.RowIndex; columna = e.ColumnIndex;
            switch (columna)
            {
                case 4:
                    {
                        nombre= (dataGridView1.Rows[fila].Cells[0].Value.ToString());
                        mp.Borrar(nombre,dataGridView1.Rows[fila].Cells[2].Value.ToString());
                        limpiar();
                        txtBuscar.Focus();
                    }
                    break;

                case 5:
                    {
                       
                        nombre = dataGridView1.Rows[fila].Cells[0].Value.ToString();
                        descripcion = dataGridView1.Rows[fila].Cells[1].Value.ToString();
                        precio = dataGridView1.Rows[fila].Cells[2].Value.ToString();
                        Form1 lbs = new Form1();
                        lbs.ShowDialog();
                        limpiar();
                        txtBuscar.Focus();
                    }
                    break;
            }
        }

    }
}
